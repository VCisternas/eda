server <- function(input, output) {
  
  output$main_plot <- renderPlot({
    
    kernel <- input$kernel
    bw <- input$bandwidth
    plot(df$accel, df$times)
    lines(ksmooth(df$accel, df$times, kernel, bandwidth = bw/0.25), col = "red")
  
  })
}
