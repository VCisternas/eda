plot.cont.data <- function(data_feature){

  index <- 1
    for (variable in data_feature) {
      if (is.na(variable)){data_feature[index] <- 0}
      index <- index + 1
    }
  
  mean_feature <- mean(data_feature)
  sd_feature <- sd(data_feature)
  xseq <- sort(data_feature, decreasing= FALSE)
  cumulative<-pnorm(xseq, mean(data_feature), sd(data_feature))

  
  
  par(mfrow=c(2,2))
  boxplot(data_feature)
  qqnorm(data_feature)
  hist(data_feature)
  plot(ecdf(data_feature))
  lines(xseq, cumulative, col ="blue", type = "l", ylim = range(0,1))
  
}

analyse.regression <- function(dat){
  lm.shoesize <- lm(dat$Schuhgroesse ~ dat$Groesse, data=dat)
  print(lm.shoesize)
  return(anova(lm.shoesize))
}
analyse.regression(data)

  1. Es gibt einen leichten linearen Zusammenhang zwischen Krper und Schuhgre. Der Zusammenhang in der Gruppe der Mnner ist leicht grer, als der in der Gruppe der Frauen.

Der Wert SSR (Regression Sums of Squeres) also der Indikator fr die Streuung der durch die Regressionsgerade geschtzten Y-Werte betrgt bei den Frauen im Durchschnitt 1.20 und bei den Mnnern 1.64. Das Bedeutet, dass die Regressionsgerade der Mnner Steiler als die der Frauen ist.

Der Wert SSE (Error Sum of Squeres) bwz. Resudien betrgt bei den Frauen 1.2 und bei den Mnnern 1.87. Das Bedeutet, dass die die Regressionsgerade der Frauen genauer als die der Mnner ist (Basierend auf der Stichprobe).
